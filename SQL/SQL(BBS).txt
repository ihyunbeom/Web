CREATE TABLE USER(
userId varchar(20),
userEmail varchar(20),
userName varchar(20),
userGender varchar(10),
userTel varchar(20),
userGrade int,
userCreated varchar(50),
PRIMARY KEY(userId)
);

CREATE TABLE PW(
userId varchar(20),
userPassword varchar(20),
PRIMARY KEY(userId)
);

CREATE TABLE BBS(
bbsId int(11),
bbsTitle varchar(50),
userEmail varchar(20),
bbsCreated varchar(50),
bbsContent varchar(2048),
bbsAvailable int,
PRIMARY KEY(bbsId)
);

CREATE TABLE REPLY(
replyId int(11),
bbsId int(11),
userEmail varchar(20),
bbsCreated varchar(50),
bbsContent varchar(2048),
bbsAvailable int,
PRIMARY KEY(replyId)
);
